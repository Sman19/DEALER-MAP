DEALER MAP function addPoint(lat, lng, name, state='', code='', phone='') { const m = L.circleMarker([lat, lng], { radius: 5, weight: 1, opacity: 1, fillOpacity: 0.85 }).bindPopup(`
${name}
State: ${state}
Dealer Code: ${code}
Phone: ${phone}
`); m.dealerName = name.toLowerCase(); allMarkers.push(m); bounds.extend([lat, lng]); }
